function Particles({coherence}){const canvasRef=useRef(null);const cohRef=useRef(coherence);useEffect(()=>{cohRef.current=coherence;},[coherence]);useEffect(()=>{const canvas=canvasRef.current;if(!canvas)return;const ctx=canvas.getContext("2d");if(!ctx)return;let frameId;const mobile=window.innerWidth<768;const particles=[];function spawn(){const W=canvas.width,H=canvas.height,cx=W/2,cy=H/2,a=Math.random()*Math.PI*2,r=14+Math.random()*52,spd=0.14+Math.random()*0.38,oa=a+(Math.random()-0.5)*1.1;particles.push({x:cx+Math.cos(a)*r,y:cy+Math.sin(a)*r,vx:Math.cos(oa)*spd,vy:Math.sin(oa)*spd-0.11,life:0,maxLife:175+Math.random()*135,size:0.5+Math.random()*1.55,hue:28+Math.random()*20});}
function draw(){frameId=requestAnimationFrame(draw);const coh=cohRef.current;ctx.clearRect(0,0,canvas.width,canvas.height);const spawnRate=mobile?0.03+coh*0.11:0.055+coh*0.20;if(Math.random()<spawnRate)spawn();for(let i=particles.length-1;i>=0;i--){const p=particles[i];p.life++;if(p.life>=p.maxLife){particles.splice(i,1);continue;}p.x+=p.vx;p.y+=p.vy;p.vx+=(Math.random()-0.5)*0.022;const prog=p.life/p.maxLife,alpha=Math.sin(prog*Math.PI)*(0.52+coh*0.32);ctx.beginPath();ctx.arc(p.x,p.y,p.size*(1+prog*0.48),0,Math.PI*2);ctx.fillStyle="hsla("+p.hue+",86%,64%,"+alpha+")";ctx.fill();}}draw();return()=>cancelAnimationFrame(frameId);},[]);return(<canvas ref={canvasRef} width={800} height={800} style={{position:"absolute" as const,inset:"-15%",width:"130%",height:"130%",pointerEvents:"none" as const,zIndex:3}}/>);}
function Ripples({triggered}){const canvasRef=useRef(null);const ripples=useRef([]);const prev=useRef(false);useEffect(()=>{if(triggered&&!prev.current){ripples.current.push({r:0,maxR:400,alpha:0.62});setTimeout(()=>ripples.current.push({r:0,maxR:305,alpha:0.40}),175);setTimeout(()=>ripples.current.push({r:0,maxR:210,alpha:0.24}),350);}prev.current=triggered;},[triggered]);useEffect(()=>{const canvas=canvasRef.current;if(!canvas)return;const ctx=canvas.getContext("2d");if(!ctx)return;let frameId;function draw(){frameId=requestAnimationFrame(draw);ctx.clearRect(0,0,canvas.width,canvas.height);const cx=canvas.width/2,cy=canvas.height/2;for(let i=ripples.current.length-1;i>=0;i--){const rip=ripples.current[i];rip.r+=2.6;rip.alpha*=0.967;if(rip.r>rip.maxR||rip.alpha<0.005){ripples.current.splice(i,1);continue;}ctx.strokeStyle="rgba(255,87,34,"+rip.alpha+")";ctx.lineWidth=1.7;ctx.beginPath();ctx.arc(cx,cy,rip.r,0,Math.PI*2);ctx.stroke();}}draw();return()=>cancelAnimationFrame(frameId);},[]);return(<canvas ref={canvasRef} width={1000} height={1000} style={{position:"absolute" as const,inset:"-28%",width:"156%",height:"156%",pointerEvents:"none" as const,zIndex:4}}/>);}

function CursorWarmth({heartPos}:{heartPos:{x:number,y:number}}){const[mouse,setMouse]=useState({x:-999,y:-999});const[near,setNear]=useState(false);useEffect(()=>{function m(e){const mx=e.clientX,my=e.clientY;setMouse({x:mx,y:my});setNear(Math.sqrt((mx-heartPos.x)**2+(my-heartPos.y)**2)<200);}window.addEventListener("mousemove",m);return()=>window.removeEventListener("mousemove",m);},[heartPos]);if(!near)return null;return(<div style={{position:"fixed" as const,left:mouse.x-70,top:mouse.y-70,width:"140px",height:"140px",borderRadius:"9999px",background:"radial-gradient(ellipse 50% 50% at 50% 50%, rgba(220,110,20,0.16) 0%, rgba(200,80,10,0.06) 55%, rgba(10,4,1,0) 100%)",pointerEvents:"none" as const,zIndex:20}}/>);}

function TransmissionRings({pulseRingsRef}:{pulseRingsRef:React.MutableRefObject<{r:number,alpha:number}[]>}){const canvasRef=useRef(null);useEffect(()=>{const canvas=canvasRef.current;if(!canvas)return;const ctx=canvas.getContext('2d');if(!ctx)return;let frameId;function resize(){canvas.width=window.innerWidth;canvas.height=window.innerHeight;}resize();window.addEventListener('resize',resize);function draw(){frameId=requestAnimationFrame(draw);ctx.clearRect(0,0,canvas.width,canvas.height);const cx=canvas.width/2;const cy=canvas.height*0.52;const maxDim=Math.max(canvas.width,canvas.height);const rings=pulseRingsRef.current;for(let i=rings.length-1;i>=0;i--){const pr=rings[i];pr.r+=11;pr.alpha*=0.971;if(pr.r>maxDim*1.8||pr.alpha<0.004){rings.splice(i,1);continue;}ctx.beginPath();ctx.arc(cx,cy,pr.r,0,Math.PI*2);ctx.strokeStyle='rgba(240,192,64,'+pr.alpha+')';ctx.lineWidth=3.5;ctx.stroke();}}draw();return()=>{cancelAnimationFrame(frameId);window.removeEventListener('resize',resize);};},[]);return<canvas ref={canvasRef} style={{position:'fixed' as const,inset:0,width:'100%',height:'100%',zIndex:55,pointerEvents:'none' as const}}/>;}
