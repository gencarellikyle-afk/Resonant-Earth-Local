function useFieldIdentity(){const[identity,setIdentity]=useState(null);useEffect(()=>{try{const stored=localStorage.getItem("re_field_id");if(stored){const name=localStorage.getItem("re_field_name")||null;const count=parseInt(localStorage.getItem("re_signal_count")||"0");const last=localStorage.getItem("re_last_signal")||"";const firstSeen=localStorage.getItem("re_first_seen")||"";setIdentity({id:stored,name,count,last,firstSeen});}}catch(e){}},[]);function claimIdentity(signalText){try{let id=localStorage.getItem("re_field_id");if(!id){id=crypto.randomUUID?crypto.randomUUID():"re-"+Date.now()+"-"+Math.random().toString(36).slice(2);localStorage.setItem("re_field_id",id);localStorage.setItem("re_first_seen",new Date().toISOString());}const count=parseInt(localStorage.getItem("re_signal_count")||"0")+1;localStorage.setItem("re_signal_count",String(count));localStorage.setItem("re_last_signal",signalText.slice(0,120));localStorage.setItem("re_last_seen",new Date().toISOString());const name=localStorage.getItem("re_field_name")||null;setIdentity({id,name,count,last:signalText.slice(0,120),firstSeen:localStorage.getItem("re_first_seen")||""});return name;}catch(e){return null;}}function setFieldName(name){try{const clean=name.trim().slice(0,48);if(!clean)return;localStorage.setItem("re_field_name",clean);setIdentity((prev)=>prev?{...prev,name:clean}:prev);}catch(e){}}return{identity,claimIdentity,setFieldName};}
