function InteractiveGlobe({coherence,ceremonyRef}){const canvasRef=useRef(null);const cohRef=useRef(coherence);const dragRef=useRef({active:false,lastX:0,lastY:0,rotX:0.3,rotY:0,velX:0,velY:0});const zoomRef=useRef(1);const realPointsRef=useRef([]);const isMobile=useRef(false);useEffect(()=>{isMobile.current=window.innerWidth<768;},[]);useEffect(()=>{supabase.from("signals").select("lat,lng,coherence").not("lat","is",null).order("created_at",{ascending:false}).limit(40).then(({data})=>{if(data)realPointsRef.current=data.filter(d=>d.lat&&d.lng).map(d=>({lat:d.lat*Math.PI/180,lon:d.lng*Math.PI/180,pulse:Math.random()*Math.PI*2,size:1.5+d.coherence*2,hot:d.coherence}));});const ch=supabase.channel("globe-signals").on("postgres_changes",{event:"INSERT",schema:"public",table:"signals"},(p)=>{if(p.new.lat&&p.new.lng)realPointsRef.current=[{lat:p.new.lat*Math.PI/180,lon:p.new.lng*Math.PI/180,pulse:Math.random()*Math.PI*2,size:1.5+p.new.coherence*2,hot:p.new.coherence},...realPointsRef.current].slice(0,40);}).subscribe();return()=>{supabase.removeChannel(ch);};},[]);useEffect(()=>{cohRef.current=coherence;},[coherence]);useEffect(()=>{const canvas=canvasRef.current;if(!canvas)return;const ctx=canvas.getContext("2d");if(!ctx)return;let frameId;const mobile=isMobile.current;const canvasSize=mobile?600:900;canvas.width=canvasSize;canvas.height=canvasSize;const latLines=mobile?12:21;const lonLines=mobile?18:32;const stars=Array.from({length:mobile?180:300},()=>({x:Math.random()*canvasSize,y:Math.random()*canvasSize,s:Math.random()*1.4+0.3,a:Math.random()*0.6+0.2}));const sigPoints=Array.from({length:24},(_,i)=>({lat:(i/24*Math.PI*2-Math.PI/2)*(0.8+Math.random()*0.4)%(Math.PI/2),lon:i/24*Math.PI*2+Math.random()*0.8,pulse:Math.random()*Math.PI*2,size:1+Math.random()*1.5,hot:Math.random()}));const pulseRings=[];function project(lat,lon,rotX,rotY,R){const x0=Math.cos(lat)*Math.sin(lon),y0=Math.sin(lat),z0=Math.cos(lat)*Math.cos(lon);const x1=x0*Math.cos(rotY)-z0*Math.sin(rotY),z1=x0*Math.sin(rotY)+z0*Math.cos(rotY);const y2=y0*Math.cos(rotX)-z1*Math.sin(rotX),z2=y0*Math.sin(rotX)+z1*Math.cos(rotX);return{x:x1*R,y:y2*R,z:z2,visible:z2>-0.15};}
function draw(){frameId=requestAnimationFrame(draw);const dr=dragRef.current;if(!dr.active){dr.rotY+=dr.velX*0.95+0.0008;dr.rotX+=dr.velY*0.95;dr.velX*=0.92;dr.velY*=0.92;dr.rotX=Math.max(-1.2,Math.min(1.2,dr.rotX));}const W=canvas.width,H=canvas.height,cx=W/2,cy=H/2,R=Math.min(W,H)*0.38*zoomRef.current,coh=cohRef.current;ctx.clearRect(0,0,W,H);stars.forEach((s)=>{ctx.beginPath();ctx.arc(s.x,s.y,s.s,0,Math.PI*2);ctx.fillStyle="rgba(255,240,220,"+s.a+")";ctx.fill();});for(let l=6;l>=0;l--){const gr=R*(1.7+l*0.45+coh*0.25),a=(0.02+coh*0.035)/(l+1);const g=ctx.createRadialGradient(cx,cy,R*0.3,cx,cy,gr);g.addColorStop(0,"rgba(210,130,30,"+(a*4)+")");g.addColorStop(0.4,"rgba(160,75,15,"+(a*2)+")");g.addColorStop(1,"rgba(80,30,5,0)");ctx.fillStyle=g;ctx.beginPath();ctx.arc(cx,cy,gr,0,Math.PI*2);ctx.fill();}for(let ai=0;ai<3;ai++){const atmGrad=ctx.createRadialGradient(cx,cy,R*(0.92+ai*0.04),cx,cy,R*(1.12+ai*0.08));atmGrad.addColorStop(0,"rgba(80,140,255,0)");atmGrad.addColorStop(0.5,"rgba(60,120,255,"+(0.07+coh*0.06-ai*0.02)+")");atmGrad.addColorStop(1,"rgba(20,60,180,0)");ctx.fillStyle=atmGrad;ctx.beginPath();ctx.arc(cx,cy,R*(1.12+ai*0.08),0,Math.PI*2);ctx.fill();}for(let aurora=0;aurora<2;aurora++){const ay=aurora===0?-R*0.82:R*0.82;const aColors=aurora===0?["rgba(255,180,40,","rgba(220,120,20,"]:["rgba(200,100,15,","rgba(255,160,30,"];for(let i=0;i<4;i++){const aSize=R*(0.28+i*0.12);const ag=ctx.createRadialGradient(cx,cy+ay,0,cx,cy+ay,aSize);ag.addColorStop(0,aColors[0]+(0.05+coh*0.04-i*0.01)+")");ag.addColorStop(0.6,aColors[1]+(0.02+coh*0.02-i*0.005)+")");ag.addColorStop(1,"rgba(0,100,200,0)");ctx.fillStyle=ag;ctx.beginPath();ctx.arc(cx,cy+ay,aSize,0,Math.PI*2);ctx.fill();}}const termX=cx+R*0.3;const termGrad=ctx.createLinearGradient(termX-R*0.15,cy,termX+R*0.15,cy);termGrad.addColorStop(0,"rgba(8,6,4,0)");termGrad.addColorStop(1,"rgba(8,6,4,0.35)");ctx.fillStyle=termGrad;ctx.beginPath();ctx.arc(cx,cy,R,0,Math.PI*2);ctx.fill();if(!mobile){for(let i=0;i<=latLines;i++){const lat=(i/latLines)*Math.PI-Math.PI/2,pts=[];for(let j=0;j<=80;j++){const p=project(lat,(j/80)*Math.PI*2,dr.rotX,dr.rotY,R);pts.push(p);}ctx.strokeStyle="rgba(220,160,55,"+(0.14+coh*0.13)+")";ctx.lineWidth=0.7;ctx.beginPath();let s=false;pts.forEach((p)=>{if(p.visible){if(!s){ctx.moveTo(cx+p.x,cy+p.y);s=true;}else ctx.lineTo(cx+p.x,cy+p.y);}else s=false;});ctx.stroke();}}if(!mobile){for(let j=0;j<lonLines;j++){const lon=(j/lonLines)*Math.PI*2,pts=[];for(let i=0;i<=60;i++){const p=project((i/60)*Math.PI-Math.PI/2,lon,dr.rotX,dr.rotY,R);pts.push(p);}ctx.beginPath();let s=false;pts.forEach((p)=>{if(p.visible){ctx.strokeStyle="rgba(220,160,55,"+(Math.max(0,p.z)*(0.12+coh*0.13))+")";ctx.lineWidth=0.7;if(!s){ctx.moveTo(cx+p.x,cy+p.y);s=true;}else ctx.lineTo(cx+p.x,cy+p.y);}else s=false;});ctx.stroke();}}if(Math.random()<0.003+coh*0.004){const src=sigPoints[Math.floor(Math.random()*sigPoints.length)];pulseRings.push({lat:src.lat,lon:src.lon,r:0,maxR:0.015+Math.random()*0.02,alpha:0.5+Math.random()*0.4,speed:0.000008+Math.random()*0.000012});}for(let ri=pulseRings.length-1;ri>=0;ri--){const ring=pulseRings[ri];ring.r+=ring.speed||0.000008;ring.alpha*=0.996;if(ring.r>ring.maxR||ring.alpha<0.008){pulseRings.splice(ri,1);continue;}const ringPts=[];for(let k=0;k<=48;k++){const a2=ring.r;const rlat=ring.lat+Math.cos(k/48*Math.PI*2)*a2;const rlon=ring.lon+Math.sin(k/48*Math.PI*2)*a2;const p=project(rlat,rlon,dr.rotX,dr.rotY,R);ringPts.push(p);}ctx.beginPath();let rs=false;ringPts.forEach((p)=>{if(p.visible){if(!rs){ctx.moveTo(cx+p.x,cy+p.y);rs=true;}else ctx.lineTo(cx+p.x,cy+p.y);}else rs=false;});ctx.strokeStyle="rgba(255,180,60,"+ring.alpha+")";ctx.lineWidth=1.2;ctx.stroke();}if(ceremonyRef&&ceremonyRef.current>0){ceremonyRef.current=Math.max(0,ceremonyRef.current-0.012);const cf=ceremonyRef.current;for(let i=0;i<=latLines;i++){const lat=(i/latLines)*Math.PI-Math.PI/2,pts=[];for(let j=0;j<=80;j++){const p=project(lat,(j/80)*Math.PI*2,dr.rotX,dr.rotY,R);pts.push(p);}ctx.strokeStyle="rgba(222,188,75,"+cf*0.9+")";ctx.lineWidth=1.2;ctx.beginPath();let s=false;pts.forEach((p)=>{if(p.visible){if(!s){ctx.moveTo(cx+p.x,cy+p.y);s=true;}else ctx.lineTo(cx+p.x,cy+p.y);}else s=false;});ctx.stroke();}for(let j=0;j<lonLines;j++){const lon=(j/lonLines)*Math.PI*2,pts=[];for(let i=0;i<=60;i++){const p=project((i/60)*Math.PI-Math.PI/2,lon,dr.rotX,dr.rotY,R);pts.push(p);}ctx.beginPath();let s=false;pts.forEach((p)=>{if(p.visible){ctx.strokeStyle="rgba(222,188,75,"+cf*0.8+")";ctx.lineWidth=1.2;if(!s){ctx.moveTo(cx+p.x,cy+p.y);s=true;}else ctx.lineTo(cx+p.x,cy+p.y);}else s=false;});ctx.stroke();}const ringR=R*(1.02+cf*0.15);ctx.beginPath();ctx.arc(cx,cy,ringR,0,Math.PI*2);ctx.strokeStyle="rgba(222,188,75,"+cf*0.7+")";ctx.lineWidth=2;ctx.stroke();}const activePoints=realPointsRef.current.length>0?realPointsRef.current:sigPoints;activePoints.forEach((pt)=>{pt.pulse+=0.04;const p=project(pt.lat,pt.lon,dr.rotX,dr.rotY,R);if(!p.visible)return;const px=cx+p.x,py=cy+p.y,glow=Math.abs(Math.sin(pt.pulse)),alpha=(0.45+glow*0.55+coh*0.3+pt.hot*0.2)*Math.max(0,p.z+0.3),size=pt.size*(0.8+glow*0.5+coh*0.3+pt.hot*0.3);const gr2=ctx.createRadialGradient(px,py,0,px,py,size*4);gr2.addColorStop(0,"rgba(255,200,80,"+alpha+")");gr2.addColorStop(0.4,"rgba(200,100,30,"+(alpha*0.5)+")");gr2.addColorStop(1,"rgba(150,60,10,0)");ctx.fillStyle=gr2;ctx.beginPath();ctx.arc(px,py,size*4,0,Math.PI*2);ctx.fill();ctx.fillStyle="rgba(255,220,100,"+Math.min(1,alpha*1.2)+")";ctx.beginPath();ctx.arc(px,py,size,0,Math.PI*2);ctx.fill();if(glow>0.6||pt.hot>0.7){ctx.strokeStyle="rgba(255,180,50,"+(glow*alpha*0.6)+")";ctx.lineWidth=1.5;ctx.beginPath();ctx.arc(px,py,size*(1+glow*4),0,Math.PI*2);ctx.stroke();}});}
draw();function md(e){const r=canvas.getBoundingClientRect();dragRef.current.active=true;dragRef.current.lastX=e.clientX-r.left;dragRef.current.lastY=e.clientY-r.top;dragRef.current.velX=0;dragRef.current.velY=0;}function mm(e){if(!dragRef.current.active)return;const r=canvas.getBoundingClientRect();const dx=e.clientX-r.left-dragRef.current.lastX,dy=e.clientY-r.top-dragRef.current.lastY;dragRef.current.velX=dx*0.003;dragRef.current.velY=dy*0.003;dragRef.current.rotY+=dx*0.004;dragRef.current.rotX+=dy*0.004;dragRef.current.rotX=Math.max(-1.2,Math.min(1.2,dragRef.current.rotX));dragRef.current.lastX=e.clientX-r.left;dragRef.current.lastY=e.clientY-r.top;}function mu(){dragRef.current.active=false;}function mw(e){e.preventDefault();zoomRef.current=Math.max(0.5,Math.min(2.5,zoomRef.current-e.deltaY*0.001));}function ts(e){const t=e.touches[0];dragRef.current.active=true;dragRef.current.lastX=t.clientX;dragRef.current.lastY=t.clientY;dragRef.current.velX=0;dragRef.current.velY=0;}function tm(e){if(!dragRef.current.active)return;const t=e.touches[0];const dx=t.clientX-dragRef.current.lastX,dy=t.clientY-dragRef.current.lastY;dragRef.current.velX=dx*0.003;dragRef.current.velY=dy*0.003;dragRef.current.rotY+=dx*0.004;dragRef.current.rotX+=dy*0.004;dragRef.current.rotX=Math.max(-1.2,Math.min(1.2,dragRef.current.rotX));dragRef.current.lastX=t.clientX;dragRef.current.lastY=t.clientY;}function te(){dragRef.current.active=false;}canvas.addEventListener("mousedown",md);window.addEventListener("mousemove",mm);window.addEventListener("mouseup",mu);canvas.addEventListener("wheel",mw,{passive:false});canvas.addEventListener("touchstart",ts,{passive:true});window.addEventListener("touchmove",tm,{passive:true});window.addEventListener("touchend",te);return()=>{cancelAnimationFrame(frameId);canvas.removeEventListener("mousedown",md);window.removeEventListener("mousemove",mm);window.removeEventListener("mouseup",mu);canvas.removeEventListener("wheel",mw);canvas.removeEventListener("touchstart",ts);window.removeEventListener("touchmove",tm);window.removeEventListener("touchend",te);};},[]);return(<canvas ref={canvasRef} width={900} height={900} className="globe-canvas" style={{position:"fixed",top:"50%",left:"50%",transform:"translate(-50%,-50%)",width:"120vw",height:"120vw",zIndex:1,cursor:"grab",opacity:0.45}}/>);}
