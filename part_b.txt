
function Particles({coherence}){const canvasRef=useRef(null);const cohRef=useRef(coherence);useEffect(()=>{cohRef.current=coherence;},[coherence]);useEffect(()=>{const canvas=canvasRef.current;if(!canvas)return;const ctx=canvas.getContext("2d");if(!ctx)return;let frameId;const particles=[];function spawn(){const W=canvas.width,H=canvas.height,cx=W/2,cy=H/2,a=Math.random()*Math.PI*2,r=20+Math.random()*60,spd=0.2+Math.random()*0.5,oa=a+(Math.random()-0.5)*1.5;particles.push({x:cx+Math.cos(a)*r,y:cy+Math.sin(a)*r,vx:Math.cos(oa)*spd,vy:Math.sin(oa)*spd-0.15,life:0,maxLife:160+Math.random()*120,size:0.6+Math.random()*1.8,hue:30+Math.random()*30});}
function draw(){frameId=requestAnimationFrame(draw);const coh=cohRef.current;ctx.clearRect(0,0,canvas.width,canvas.height);if(Math.random()<0.08+coh*0.28)spawn();for(let i=particles.length-1;i>=0;i--){const p=particles[i];p.life++;if(p.life>=p.maxLife){particles.splice(i,1);continue;}p.x+=p.vx;p.y+=p.vy;p.vx+=(Math.random()-0.5)*0.03;const prog=p.life/p.maxLife,alpha=Math.sin(prog*Math.PI)*(0.6+coh*0.35);ctx.beginPath();ctx.arc(p.x,p.y,p.size*(1+prog*0.6),0,Math.PI*2);ctx.fillStyle="hsla("+p.hue+",90%,68%,"+alpha+")";ctx.fill();}}draw();return()=>cancelAnimationFrame(frameId);},[]);
return(<canvas ref={canvasRef} width={800} height={800} style={{position:"absolute",inset:"-15%",width:"130%",height:"130%",pointerEvents:"none",zIndex:3}}/>);}
function Ripples({triggered}){const canvasRef=useRef(null);const ripples=useRef([]);const prev=useRef(false);useEffect(()=>{if(triggered&&!prev.current){ripples.current.push({r:0,maxR:380,alpha:0.7});setTimeout(()=>ripples.current.push({r:0,maxR:290,alpha:0.45}),160);setTimeout(()=>ripples.current.push({r:0,maxR:200,alpha:0.28}),320);}prev.current=triggered;},[triggered]);useEffect(()=>{const canvas=canvasRef.current;if(!canvas)return;const ctx=canvas.getContext("2d");if(!ctx)return;let frameId;function draw(){frameId=requestAnimationFrame(draw);ctx.clearRect(0,0,canvas.width,canvas.height);const cx=canvas.width/2,cy=canvas.height/2;for(let i=ripples.current.length-1;i>=0;i--){const rip=ripples.current[i];rip.r+=3;rip.alpha*=0.965;if(rip.r>rip.maxR||rip.alpha<0.006){ripples.current.splice(i,1);continue;}ctx.strokeStyle="rgba(255,150,50,"+rip.alpha+")";ctx.lineWidth=2;ctx.beginPath();ctx.arc(cx,cy,rip.r,0,Math.PI*2);ctx.stroke();}}draw();return()=>cancelAnimationFrame(frameId);},[]);
return(<canvas ref={canvasRef} width={900} height={900} style={{position:"absolute",inset:"-25%",width:"150%",height:"150%",pointerEvents:"none",zIndex:4}}/>);}
